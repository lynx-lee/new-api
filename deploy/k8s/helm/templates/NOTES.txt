1. Get the application URL by running these commands:

  {{- range .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ .host }}/
  {{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "new-api.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "New-API URL: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch its status by running 'kubectl get svc -w {{ include "new-api.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "new-api.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "New-API URL: http://$SERVICE_PORT:{{ $svcPort }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  Run these commands to port-forward the service:
  kubectl port-forward service/{{ include "new-api.fullname" . }} {{ $svcPort }}:{{ $svcPort }}
  echo "New-API URL: http://127.0.0.1:{{ $svcPort }}"
{{- end }}
